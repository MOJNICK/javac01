CREATE TABLE SKALAOCEN(
ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
OCENA VARCHAR(3) NOT NULL,
CONSTRAINT SKALAOCEN_PK PRIMARY KEY(ID)
);

CREATE TABLE PREDMIOT(
ID INTEGER,
NAZWA VARCHAR(10) NOT NULL
);

CREATE TABLE STUDENT(
ID INTEGER,
NRALBUMU VARCHAR(10),
IMIE VARCHAR(10) NOT NULL,
NAZWISKO VARCHAR(10) NOT NULL
);

CREATE TABLE OCENA(
ID INTEGER,
DZIEN DATE,
STUDNET_ID INTEGER NOT NULL,
PRZEDMIOT_ID INTEGER NOT NULL,
SKALAOCEN_ID INTEGER NOT NULL,
FOREIGN KEY (ID) REFERENCES SKALAOCEN(ID),
);
