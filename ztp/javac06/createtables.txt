CREATE TABLE SKALAOCEN(
ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
OCENA VARCHAR(3) NOT NULL,
CONSTRAINT SKALAOCEN_PK PRIMARY KEY (ID)
);

CREATE TABLE PRZEDMIOT(
ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
NAZWA VARCHAR(10) NOT NULL,
CONSTRAINT PRZEDMIOT_PK PRIMARY KEY (ID)
);

CREATE TABLE STUDENT(
ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
NRALBUMU VARCHAR(10),
IMIE VARCHAR(10) NOT NULL,
NAZWISKO VARCHAR(10) NOT NULL,
CONSTRAINT STUDENT_PK PRIMARY KEY (ID)
);

CREATE TABLE OCENA(
ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
DZIEN DATE NOT NULL,
STUDENT_ID INTEGER,
PRZEDMIOT_ID INTEGER,
SKALAOCEN_ID INTEGER,
CONSTRAINT OCENA_PK  PRIMARY KEY (ID),
CONSTRAINT OCENA_FK1 FOREIGN KEY (STUDENT_ID) REFERENCES STUDENT(ID),
CONSTRAINT OCENA_FK2 FOREIGN KEY (PRZEDMIOT_ID) REFERENCES PRZEDMIOT(ID),
CONSTRAINT OCENA_FK3 FOREIGN KEY (SKALAOCEN_ID) REFERENCES SKALAOCEN(ID)
);
